дёҖгҖҒAnnotationConfigApplicationContext е…ҘеҸЈ

дҫқиө–дәҺ Spring5.X зүҲжң¬пјҢзӣ®еүҚеӨ§еӨҡж•°йЎ№зӣ®е·Іж”ҫејғеҹәдәҺ XML й…ҚзҪ®зҡ„ж–№ејҸпјҢжүҖд»Ҙжң¬ж–ҮеҹәдәҺ JavaConfig жіЁи§Јй…ҚзҪ®зҡ„ж–№ејҸзҗҶи§Ј Spring дёҠдёӢж–ҮеҗҜеҠЁзҡ„жөҒзЁӢгҖӮ

AnnotationConfigApplicationContext context = new AnnotationConfigApplicationContext(Config.class);
// жҲ–
AnnotationConfigApplicationContext context = new AnnotationConfigApplicationContext();

дәҢгҖҒAnnotationConfigApplicationContext жһ„йҖ ж–№жі•

ж— и®әйҖҡиҝҮе“Әз§Қж–№ејҸжңҖз»ҲйғҪдјҡи°ғз”Ёй»ҳи®Өжһ„йҖ еҷЁгҖӮ

public AnnotationConfigApplicationContext(Class<?>... componentClasses) {
	// е…Ҳи°ғз”ЁеҪ“еүҚзұ»зҡ„жһ„йҖ еҷЁпјҢеҲқе§ӢеҢ–BeanDefinitionзҡ„иҜ»еҸ–еҷЁгҖҒжү«жҸҸеҷЁ
	this();
	// жіЁеҶҢй…ҚзҪ®зұ»пјҢжң¬иә«д№ҹдҪңдёәдёҖдёӘBeanDefinitionиў«жіЁеҶҢ
	register(componentClasses);
	// еҲ·ж–°ж–№жі•
	refresh();
}

й»ҳи®Өжһ„йҖ еҷЁпјҢдјҳе…Ҳи°ғз”ЁзҲ¶зұ» GenericApplicationContext зҡ„жһ„йҖ еҷЁгҖӮ

public AnnotationConfigApplicationContext() {
	// дјҳе…Ҳи°ғз”ЁзҲ¶зұ»зҡ„жһ„йҖ еҷЁпјҢGenericApplicationContextпјҢеҲқе§ӢеҢ–BeanFactory
	// еҲқе§ӢеҢ–иў«жіЁи§Јзҡ„BeanDefinitionиҜ»еҸ–еҷЁ
	this.reader = new AnnotatedBeanDefinitionReader(this);
	// еҲқе§ӢеҢ–ClassPathдёӢBeanDefinitionзҡ„жү«жҸҸеҷЁ
	// иҝҷйҮҢзҡ„ scanner д»…д»…жҳҜжҸҗдҫӣз»ҷзЁӢеәҸе‘ҳеӨ–йғЁи°ғз”Ёзҡ„пјҢSpring еҶ…йғЁжү«жҸҸеҢ…дҪҝз”Ёзҡ„жҳҜдёӢйқўж–№жі•дёӯзҡ„ scanner
	// org.springframework.context.annotation.ComponentScanAnnotationParser#parse
	this.scanner = new ClassPathBeanDefinitionScanner(this);
}

public GenericApplicationContext() {
	this.beanFactory = new DefaultListableBeanFactory();
}

AnnotationConfigApplicationContext й»ҳи®Өжһ„йҖ еҷЁдёӯеҒҡдәҶдёӨд»¶дәӢпјҡ

  • еҲқе§ӢеҢ– AnnotatedBeanDefinitionReader BeanDefinitionиҜ»еҸ–еҷЁ
  • еҲқе§ӢеҢ– ClassPathBeanDefinitionScanner д»…дҫӣзЁӢеәҸе‘ҳеӨ–йғЁи°ғз”ЁпјҢе№¶дёҚжҳҜ Spring еҶ…йғЁжү«жҸҸеҢ…зҡ„зұ»

2.1 еҲқе§ӢеҢ– AnnotatedBeanDefinitionReader

дёә BeanFactory жіЁеҶҢеҗҺзҪ®еӨ„зҗҶеҷЁ

е®һдҫӢеҢ–дёҖдёӘ AnnotatedBeanDefinitionReader е§”жүҳ AnnotationConfigUtilsпјҢ registerAnnotationConfigProcessors ж–№жі•еҫҲйҮҚиҰҒпјҢеңЁиҝҷдёӘж–№жі•дёӯдёәе®№еҷЁжіЁеҶҢдёҖзі»еҲ—еҶ…йғЁзҡ„ BeanPostProcessor еҗҺзҪ®еӨ„зҗҶеҷЁпјҲ6дёӘпјүгҖӮ

  • ConfigurationClassPostProcessor: жҸҗдҫӣ Java й…ҚзҪ®зұ»еӨ„зҗҶиғҪеҠӣгҖӮ
  • AutowiredAnnotationBeanPostProcessor: жҸҗдҫӣ Autowired жіЁи§Јзҡ„BeanеҗҺзҪ®еӨ„зҗҶеҷЁеҠҹиғҪгҖӮ
  • CommonAnnotationBeanPostProcessor: жҸҗдҫӣ JSR250 ж ҮеҮҶзҡ„йҖҡз”ЁжіЁи§ЈеӨ„зҗҶеҷЁеҠҹиғҪпјҢResource жіЁи§ЈпјҢ@@PostConstructпјҢ@PreDestroy зӯүгҖӮ
  • PersistenceAnnotationBeanPostProcessor: жҸҗдҫӣ JPA зӣёе…іж”ҜжҢҒгҖӮ
  • DefaultEventListenerFactory
  • EventListenerMethodProcessor
public static Set<BeanDefinitionHolder> registerAnnotationConfigProcessors(
		BeanDefinitionRegistry registry, @Nullable Object source) {
	DefaultListableBeanFactory beanFactory = unwrapDefaultListableBeanFactory(registry);
	if (beanFactory != null) {
		// AnnotationAwareOrderComparator зҡ„дё»иҰҒеҠҹиғҪжҳҜи§Јжһҗ @Order жіЁи§Је’Ң @Priority
		if (!(beanFactory.getDependencyComparator() instanceof AnnotationAwareOrderComparator)) {
			beanFactory.setDependencyComparator(AnnotationAwareOrderComparator.INSTANCE);
		}
		// ContextAnnotationAutowireCandidateResolver жҸҗдҫӣеӨ„зҗҶе»¶иҝҹеҠ иҪҪзҡ„еҠҹиғҪ
		if (!(beanFactory.getAutowireCandidateResolver() instanceof ContextAnnotationAutowireCandidateResolver)) {
			beanFactory.setAutowireCandidateResolver(new ContextAnnotationAutowireCandidateResolver());
		}
	}
	// BeanDefinitionHolder жҳҜеҜ№beanNameе’ҢBeanDefinitionзҡ„е°ҒиЈ…
	Set<BeanDefinitionHolder> beanDefs = new LinkedHashSet<>(8);
	// ConfigurationClassPostProcessor зҡ„зұ»еһӢжҳҜ BeanDefinitionRegistryPostProcessor
	// BeanDefinitionRegistryPostProcessor жҳҜ BeanFactoryPostProcessor зҡ„еӯҗжҺҘеҸЈ
	if (!registry.containsBeanDefinition(CONFIGURATION_ANNOTATION_PROCESSOR_BEAN_NAME)) {
		RootBeanDefinition def = new RootBeanDefinition(ConfigurationClassPostProcessor.class);
		def.setSource(source);
		beanDefs.add(registerPostProcessor(registry, def, CONFIGURATION_ANNOTATION_PROCESSOR_BEAN_NAME));
	}
	// AutowiredAnnotationBeanPostProcessor жҸҗдҫӣ Autowired жіЁи§Јзҡ„BeanеҗҺзҪ®еӨ„зҗҶеҷЁеҠҹиғҪ
	if (!registry.containsBeanDefinition(AUTOWIRED_ANNOTATION_PROCESSOR_BEAN_NAME)) {
		RootBeanDefinition def = new RootBeanDefinition(AutowiredAnnotationBeanPostProcessor.class);
		def.setSource(source);
		beanDefs.add(registerPostProcessor(registry, def, AUTOWIRED_ANNOTATION_PROCESSOR_BEAN_NAME));
	}
	// CommonAnnotationBeanPostProcessor жҸҗдҫӣ JSR250 ж ҮеҮҶзҡ„йҖҡз”ЁжіЁи§ЈеӨ„зҗҶеҷЁеҠҹиғҪпјҢResource жіЁи§Ј
	// Check for JSR-250 support, and if present add the CommonAnnotationBeanPostProcessor.
	if (jsr250Present && !registry.containsBeanDefinition(COMMON_ANNOTATION_PROCESSOR_BEAN_NAME)) {
		RootBeanDefinition def = new RootBeanDefinition(CommonAnnotationBeanPostProcessor.class);
		def.setSource(source);
		beanDefs.add(registerPostProcessor(registry, def, COMMON_ANNOTATION_PROCESSOR_BEAN_NAME));
	}
	// PersistenceAnnotationBeanPostProcessor жҸҗдҫӣ JPA зӣёе…іж”ҜжҢҒ
	// Check for JPA support, and if present add the PersistenceAnnotationBeanPostProcessor.
	if (jpaPresent && !registry.containsBeanDefinition(PERSISTENCE_ANNOTATION_PROCESSOR_BEAN_NAME)) {
		RootBeanDefinition def = new RootBeanDefinition();
		try {
			def.setBeanClass(ClassUtils.forName(PERSISTENCE_ANNOTATION_PROCESSOR_CLASS_NAME,
					AnnotationConfigUtils.class.getClassLoader()));
		} catch (ClassNotFoundException ex) {
			throw new IllegalStateException(
					"Cannot load optional framework class: " + PERSISTENCE_ANNOTATION_PROCESSOR_CLASS_NAME, ex);
		}
		def.setSource(source);
		beanDefs.add(registerPostProcessor(registry, def, PERSISTENCE_ANNOTATION_PROCESSOR_BEAN_NAME));
	}
	if (!registry.containsBeanDefinition(EVENT_LISTENER_PROCESSOR_BEAN_NAME)) {
		RootBeanDefinition def = new RootBeanDefinition(EventListenerMethodProcessor.class);
		def.setSource(source);
		beanDefs.add(registerPostProcessor(registry, def, EVENT_LISTENER_PROCESSOR_BEAN_NAME));
	}
	if (!registry.containsBeanDefinition(EVENT_LISTENER_FACTORY_BEAN_NAME)) {
		RootBeanDefinition def = new RootBeanDefinition(DefaultEventListenerFactory.class);
		def.setSource(source);
		beanDefs.add(registerPostProcessor(registry, def, EVENT_LISTENER_FACTORY_BEAN_NAME));
	}
	return beanDefs;
}

BeanPostProcessor еҗҺзҪ®еӨ„зҗҶеҷЁ

BeanPostProcessor жҳҜ Spring жЎҶжһ¶зҡ„дёҖдёӘжү©еұ•зӮ№пјҲдёҚжӯўдёҖдёӘпјҢ5дёӘжү©еұ•зӮ№пјүпјҢйҖҡиҝҮе®һзҺ° BeanPostProcessor жҺҘеҸЈпјҢ зЁӢеәҸе‘ҳеҸҜд»Ҙе№Ійў„ Bean зҡ„е®һдҫӢеҢ–зҡ„иҝҮзЁӢпјҢд»ҺиҖҢеҮҸиҪ»дәҶ BeanFactory зҡ„иҙҹжӢ…пјҢиҝҷдёӘжҺҘеҸЈеҸҜд»Ҙи®ҫзҪ®еӨҡдёӘпјҢдјҡеҪўжҲҗдёҖдёӘеҲ—иЎЁпјҢз„¶еҗҺдҫқж¬Ўжү§иЎҢгҖӮ жҜ”еҰӮпјҢAOP е°ұжҳҜеңЁ Bean е®һдҫӢеҢ–еҗҺжңҹе°ҶеҲҮйқўйҖ»иҫ‘з»Үе…Ҙ Bean е®һдҫӢдёӯзҡ„пјҢAOP д№ҹжӯЈжҳҜйҖҡиҝҮ BeanPostProcessor е’Ң IOC е®№еҷЁе»әз«ӢиҒ”зі»зҡ„ пјҲз”ұ Spring жҸҗдҫӣзҡ„й»ҳи®Өзҡ„ PostProcessorпјҢSpring жҸҗдҫӣдәҶеҫҲеӨҡй»ҳи®Өзҡ„ PostProcessor зҡ„е®һзҺ°зұ»пјүгҖӮ

BeanPostProcessor еёёз”Ёе®һзҺ°зұ»пјҡ
